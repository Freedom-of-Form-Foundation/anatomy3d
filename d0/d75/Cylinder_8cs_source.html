185 public override dvec3 GetNormalAt(dvec2 uv)
186 {
187 DebugUtil.AssertAllFinite(uv, nameof(uv));
188 double u = uv.x;
189 double v = uv.y;
190
191 dvec3 curveTangent = CenterCurve.GetTangentAt(v).Normalized;
192 dvec3 curveNormal = CenterCurve.GetNormalAt(v).Normalized;
193 dvec3 curveBinormal = dvec3.Cross(curveTangent, curveNormal);
194
195 double startAngle = StartAngle.GetValueAt(v);
196 double endAngle = EndAngle.GetValueAt(v);
197
198 return Math.Cos(u)*curveNormal + Math.Sin(u)*curveBinormal;
199 }

214 public override List<Vertex> GenerateVertexList(int resolutionU, int resolutionV)
215 {
216 // If asked to sample at zero points, return an empty list.
217 if (resolutionU <= 0 || resolutionV <= 0)
218 {
219 return new List<Vertex>();
220 }
221 var roughs = ParallelEnumerable.Range(0, resolutionV + 1).AsOrdered().SelectMany((j =>
222 {
223 double v = j / (double) resolutionV;
224
225 // Find the values at each ring:
226 dvec3 curveTangent = CenterCurve.GetTangentAt(v).Normalized;
227 dvec3 curveNormal = CenterCurve.GetNormalAt(v).Normalized;
228 dvec3 curveBinormal = dvec3.Cross(curveTangent, curveNormal);
229
230 dvec3 translation = CenterCurve.GetPositionAt(v);
231
232 double startAngle = StartAngle.GetValueAt(v);
233 double endAngle = EndAngle.GetValueAt(v);
234
235 return Enumerable.Range(0, resolutionU).Select((i) =>
236 {
237 double u = startAngle + (endAngle - startAngle) * i / resolutionU;
238
239 double radius = Radius.GetValueAt(new dvec2(u, v));
240
241 // Calculate the position of the rings of vertices:
242 dvec3 surfaceNormal = Math.Cos(u) * curveNormal + Math.Sin(u) * curveBinormal;
243 dvec3 surfacePosition = translation + radius * surfaceNormal;
244 return new Vertex((vec3)surfacePosition, (vec3)surfaceNormal);
245 });
246 }));
247
248 List<Vertex> output = roughs.ToList();
249
250 // Recalculate the surface normal after deformation:
251 for (int j = 1; j < resolutionV; j++)
252 {
253 for (int i = 0; i < (resolutionU - 1); i++)
254 {
255 dvec3 surfacePosition = output[(j-1)*resolutionU + i].Position;
256 dvec3 du = surfacePosition - output[(j-1)*resolutionU + i + 1].Position;
257 dvec3 dv = surfacePosition - output[(j)*resolutionU + i].Position;
258
259 // Calculate the position of the rings of vertices:
260 dvec3 surfaceNormal = dvec3.Cross(du.Normalized, dv.Normalized);
261
262 output[(j-1)*resolutionU + i] = new Vertex((vec3)surfacePosition, (vec3)surfaceNormal);
263 }
264
265 // Stitch the end of the triangles:
266 dvec3 surfacePosition2 = output[(j-1)*resolutionU + resolutionU-1].Position;
267 dvec3 du2 = surfacePosition2 - output[(j-1)*resolutionU].Position;
268 dvec3 dv2 = surfacePosition2 - output[(j)*resolutionU + resolutionU-1].Position;
269
270 // Calculate the position of the rings of vertices:
271 dvec3 surfaceNormal2 = dvec3.Cross(du2.Normalized, dv2.Normalized);
272
273 output[(j-1)*resolutionU + resolutionU-1] = new Vertex((vec3)surfacePosition2, (vec3)surfaceNormal2);
274 }
275
276 return output;
277 }

280 public override List<int> GenerateIndexList(int resolutionU, int resolutionV, int indexOffset = 0)
281 {
282 List<int> output = new List<int>(CalculateIndexCount(resolutionU, resolutionV));
283
284 // Add the remaining rings:
285 for (int j = 0; j < resolutionV; j++)
286 {
287 // Add a ring of triangles:
288 for (int i = 0; i < resolutionU-1; i++)
289 {
290 output.Add(indexOffset + i + resolutionU*j);
291 output.Add(indexOffset + i + resolutionU*(j+1));
292 output.Add(indexOffset + (i+1) + resolutionU*j);
293
294 output.Add(indexOffset + i + resolutionU*(j+1));
295 output.Add(indexOffset + (i+1) + resolutionU*(j+1));
296 output.Add(indexOffset + (i+1) + resolutionU*j);
297 }
298
299 // Stitch the end of the triangles:
300 output.Add(indexOffset + resolutionU-1 + resolutionU*j);
301 output.Add(indexOffset + resolutionU-1 + resolutionU*(j+1));
302 output.Add(indexOffset + resolutionU*j);
303
304 output.Add(indexOffset + resolutionU-1 + resolutionU*(j+1));
305 output.Add(indexOffset + resolutionU*(j+1));
306 output.Add(indexOffset + resolutionU*j);
307 }
308
309 return output;
310 }
