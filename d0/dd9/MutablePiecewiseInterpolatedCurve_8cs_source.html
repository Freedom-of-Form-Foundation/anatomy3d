120 public DerivableFunction<double, TOut> CurrentCurve()
121 {
122 lock (_curveLock)
123 {
124 // This uses the lazily-evaluated null-coalescence operator to either return the curve we have
125 // or, if we don't have one, calculate a new one, save it, and then return that.
126 // We save the curves we calculate to avoid the (potentially expensive) recalculation every time
127 // a single value is requested, but recalculate whenever we did not have one (either because it
128 // was never calculated before, or because we dropped it since a point moved and we need to
129 // recalculate the curve next time something wants it - which is right now).
130 return _currentCurve ?? (_currentCurve = _curveFactory.NewCurve(_points));
131 }
132 }
133
137 private void BecomeDirty()
138 {
139 lock (_curveLock)
140 {
141 _currentCurve = null;
142 }
143 }
