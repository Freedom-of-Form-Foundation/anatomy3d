78 public void TestSolve(double a0, double a1, double a2, double[] want)
79 {
80 IList<double> got = QuadraticFunction.Solve(a0, a1, a2).ToList();
81 Assert.Equal(want.Length, got.Count);
82 // There are no guarantees as to the order of values returned, so sort both sequences and compare items
83 // at equal locations in the sorted sequence. Expect them all to match. We don't need to try any other
84 // permutations of these lists to find out if there's some way they do match:
85 //
86 // If the smallest unmatched elements of each set are not adequately near each other, there is no way to
87 // match the smaller of the two to any element in its other set. We have just compared it against the
88 // smallest element in the other set, and found it to be too large. It cannot be too small, because we
89 // are specifically considering the smaller element of the mismatched pair. Since the too-large element
90 // was a smallest element of its set, all other elements are at least that large. Therefore, there is no
91 // element small enough to match the smallest unmatched element between both sets, so the sets are unequal.
92 // Backtracking does not help, because to find a smaller element of the opposite set, we must take it away
93 // from its match to an element that is no larger than the one we are now matching (because all larger
94 // elements are still in the unmatched portion of the set), and now _that_ element has the same problem
95 // but worse.
96 //
97 // Therefore, there is no reason to attempt any other order than an element-by-element pairwise comparison
98 // of the sorted sequences. ∎
99 Array.Sort(want);
100 int i = 0;
101 var gotSorted = from c in got orderby c select c;
102 foreach (double f in gotSorted)
103 {
104 Assert.Equal(want[i++], f, 6);
105 }
106 }
107 }
