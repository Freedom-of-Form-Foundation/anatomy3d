125 public double RayIntersect(Ray ray)
126 {
127 dvec3 rayStart = ray.StartPosition;
128 dvec3 rayDirection = ray.Direction;
129 DebugUtil.AssertAllFinite(rayStart, nameof(rayStart));
130 DebugUtil.AssertAllFinite(rayDirection, nameof(rayDirection));
131
132 // Since we raytrace only using a cylindrical surface that is horizontal and at the origin, we
133 // first shift and rotate the ray such that we get the right orientation:
134 dvec3 start = CenterCurve.GetStartPosition();
135 dvec3 end = CenterCurve.GetEndPosition();
136 DebugUtil.AssertAllFinite(start, nameof(start));
137 DebugUtil.AssertAllFinite(end, nameof(end));
138
139 dvec3 tangent = CenterCurve.GetTangentAt(0.0).Normalized;
140 dvec3 normal = CenterCurve.GetNormalAt(0.0).Normalized;
141 dvec3 binormal = CenterCurve.GetBinormalAt(0.0).Normalized;
142 DebugUtil.AssertAllFinite(tangent, nameof(tangent));
143 DebugUtil.AssertAllFinite(normal, nameof(normal));
144 DebugUtil.AssertAllFinite(binormal, nameof(binormal));
145
146 double length = dvec3.Distance(start, end);
147 DebugUtil.AssertFinite(length, nameof(length));
148
149 // CenterCurve is guaranteed to be a LineSegment, since the base property CenterCurve is masked by this
150 // class' CenterCurve property that only accepts a LineSegment, and similarly this class' constructor only
151 // accepts a LineSegment. The following mathematics, which assumes that the central axis is a line segment,
152 // is therefore valid.
153
154 dmat3 rotationMatrix = new dmat3(normal, binormal, tangent/length).Transposed;
155
156 dvec3 rescaledRay = rotationMatrix * (rayStart - start);
157 dvec3 newDirection = rotationMatrix * rayDirection.Normalized;
158
159
160 double x0 = rescaledRay.x;
161 double y0 = rescaledRay.y;
162 double z0 = rescaledRay.z;
163
164 double a = newDirection.x;
165 double b = newDirection.y;
166 double c = newDirection.z;
167
168 // Raytrace using a cylindrical surface equation x^2 + y^2. The parameters in the following line
169 // represent the coefficients of the expanded cylindrical surface equation, after the substitution
170 // x = x_0 + a t and y = y_0 + b t:
171 QuarticFunction surfaceFunction = new QuarticFunction(x0*x0 + y0*y0, 2.0*(x0*a + y0*b), a*a + b*b, 0.0, 0.0);
172
173 IEnumerable<double> intersections = Radius.SolveRaytrace(surfaceFunction, z0, c);
174
175 // The previous function returns a list of intersection distances. The value closest to 0.0f represents the
176 // closest intersection point.
177 double minimum = Single.PositiveInfinity;
178
179 foreach (double i in intersections)
180 {
181 // Calculate the 3d point at which the ray intersects the cylinder:
182 dvec3 intersectionPoint = rayStart + i*rayDirection;
183
184 // Find the closest point to the intersectionPoint on the centerLine.
185 // Get the vector v from the start of the cylinder to the intersection point:
186 dvec3 v = intersectionPoint - start;
187
188 // ...And project this vector onto the center line:
189 double t = -dvec3.Dot(intersectionPoint, tangent*length)/(length*length);
190
191 // Now we have the parameter t on the surface of the SymmetricCylinder at which the ray intersects.
192
193 // Find the angle to the normal of the centerLine, so that we can determine whether the
194 // angle is within the bound of the pie-slice at position t:
195 dvec3 centerLineNormal = CenterCurve.GetNormalAt(t);
196 dvec3 centerLineBinormal = CenterCurve.GetBinormalAt(t);
197 dvec3 d = intersectionPoint - CenterCurve.GetPositionAt(t);
198 double correctionShift = Math.Sign(dvec3.Dot(d, centerLineBinormal));
199 double phi = (correctionShift * Math.Acos(dvec3.Dot(d, centerLineNormal))) % (2.0*Math.PI);
200
201 // Determine if the ray is inside the pie-slice of the cylinder that is being displayed,
202 // otherwise discard:
203 if ( phi > StartAngle.GetValueAt(t) && phi < EndAngle.GetValueAt(t) && i >= 0.0)
204 {
205 minimum = Math.Sign(i) * Math.Min(Math.Abs(minimum), Math.Abs(i));
206 }
207 }
208
209 return minimum;
210 }
