98 public override dvec3 GetNormalAt(dvec2 uv)
99 {
100 DebugUtil.AssertAllFinite(uv, nameof(uv));
101 double u = uv.x;
102 double v = uv.y;
103
104 if ((v >= -0.5 * Math.PI) && (v < 0.0))
105 {
106 return startCap.GetNormalAt(new dvec2(u, v + 0.5 * Math.PI));
107 }
108 else if ((v >= 0.0) && (v < 1.0))
109 {
110 return shaft.GetNormalAt(new dvec2(u, v));

113 {

148 public override List<Vertex> GenerateVertexList(int resolutionU, int resolutionV)
149 {
150 // Generate the vertices of two end caps, and a shaft in between:
151 List<Vertex> startCapList = startCap.GenerateVertexList(resolutionU, resolutionU/4);
152 List<Vertex> endCapList = endCap.GenerateVertexList(resolutionU, resolutionU/4);
153
154 // Using a slice, remove the first and last ring of vertices of the shaft, which overlap with the last
155 // rings of the startCap and endCap respectively. Later we will stitch the shaft and end caps together with
156 // triangles between them, during the GenerateIndexList() step.
157 Slice<Vertex> shaftSlice = new Slice<Vertex>(
158 shaft.GenerateVertexList(resolutionU, resolutionV),
159 resolutionU,
160 Cylinder.CalculateVertexCount(resolutionU, resolutionV) - 2*resolutionU);
161
162 // Recalculate the surface normal between the cylinder and the start cap:
163 for (int i = 0; i < (resolutionU - 1); i++)
164 {
165 dvec3 surfacePosition = startCapList[(resolutionU/4-1)*resolutionU + i + 1].Position;
166 dvec3 du = surfacePosition - startCapList[(resolutionU/4-1)*resolutionU + i + 1 + 1].Position;
167 dvec3 dv = surfacePosition - shaftSlice[i].Position;
168
169 // Calculate the position of the rings of vertices:
170 dvec3 surfaceNormal = dvec3.Cross(du.Normalized, dv.Normalized);
171
172 startCapList[(resolutionU/4-1)*resolutionU + i + 1] = new Vertex((vec3)surfacePosition, (vec3)surfaceNormal);
173 }
174
175 // Stitch the end of the triangles:
176 dvec3 surfacePosition2 = startCapList[(resolutionU/4-1)*resolutionU + resolutionU].Position;
177 dvec3 du2 = surfacePosition2 - startCapList[(resolutionU/4-1)*resolutionU + 1].Position;
178 dvec3 dv2 = surfacePosition2 - shaftSlice[resolutionU].Position;
179
180 // Calculate the position of the rings of vertices:
181 dvec3 surfaceNormal2 = dvec3.Cross(du2.Normalized, dv2.Normalized);
182
183 startCapList[(resolutionU/4-1)*resolutionU + resolutionU] = new Vertex((vec3)surfacePosition2, (vec3)surfaceNormal2);
184
185 return startCapList.Concat(shaftSlice).Concat(endCapList).ToList();
186 }

189 public override List<int> GenerateIndexList(int resolutionU, int resolutionV, int indexOffset = 0)
190 {
191 // The index list of a model tells the GPU between which vertices to draw triangles. Each three consecutive
192 // indices constitute a single triangle. Since the Capsule vertex array consists of several lists stitched
193 // together, the indices of the shaft and endCap will have shifted by some amount. Therefore, calculate
194 // this index offset for all submodels:
195 int indexOffset2 = indexOffset + Hemisphere.CalculateVertexCount(resolutionU, resolutionU/4) - resolutionU;
196 int indexOffset3 = indexOffset2 + Cylinder.CalculateVertexCount(resolutionU, resolutionV) - resolutionU;
197 int indexOffset4 = indexOffset3 + Hemisphere.CalculateVertexCount(resolutionU, resolutionU/4) - resolutionU;
198
199 // Generate the startCap indices as normal:
200 List<int> startCapList = startCap.GenerateIndexList(resolutionU, resolutionU/4, indexOffset);
201
202 // Generate a shaft, using the last ring of the startCap to begin with. This is done by subtracting
203 // `resolutionU` from indexOffset2. This automatically stitches the gap between the start cap and the shaft:
204 List<int> shaftList = shaft.GenerateIndexList(resolutionU, resolutionV-1, indexOffset2);
205
206 // Generate the endCap indices as normal:
207 List<int> endCapList = endCap.GenerateIndexList(resolutionU, resolutionU/4, indexOffset3);
208
209 // The final ring between the endCap and the shaft must be generated manually:
210 List<int> stitchList = new List<int>(resolutionU);
211
212 // Add a ring of triangles:
213 for (int i = 1; i < resolutionU-1; i++)
214 {
215 int invertedI = resolutionU - i;
216 stitchList.Add(indexOffset + invertedI + indexOffset4);
217 stitchList.Add(indexOffset + (i+1) + indexOffset3 - resolutionU);
218 stitchList.Add(indexOffset + i + indexOffset3 - resolutionU);
219
220 stitchList.Add(indexOffset + (invertedI-1) + indexOffset4);
221 stitchList.Add(indexOffset + (i+1) + indexOffset3 - resolutionU);
222 stitchList.Add(indexOffset + invertedI + indexOffset4);
223 }
224
225 // Stitch the end of the ring of triangles:
226 stitchList.Add(indexOffset + 1 + indexOffset4);
227 stitchList.Add(indexOffset + indexOffset3 - resolutionU);
228 stitchList.Add(indexOffset + resolutionU-1 + indexOffset3 - resolutionU);
229
230 stitchList.Add(indexOffset + indexOffset4);
231 stitchList.Add(indexOffset + indexOffset3 - resolutionU);
232 stitchList.Add(indexOffset + 1 + indexOffset4);
233
234 stitchList.Add(indexOffset + resolutionU-1 + indexOffset4);
235 stitchList.Add(indexOffset + 1 + indexOffset3 - resolutionU);
236 stitchList.Add(indexOffset + indexOffset3 - resolutionU);
237
238 stitchList.Add(indexOffset + indexOffset4);
239 stitchList.Add(indexOffset + resolutionU-1 + indexOffset4);
240 stitchList.Add(indexOffset + indexOffset3 - resolutionU);
241
242 return startCapList.Concat(shaftList).Concat(endCapList).Concat(stitchList).ToList();
243 }
