78 public static IEnumerable<double> Solve(double a0, double a1, double a2)
79 {
80 DebugUtil.AssertFinite(a0, nameof(a0));
81 DebugUtil.AssertFinite(a1, nameof(a1));
82 DebugUtil.AssertFinite(a2, nameof(a2));
83 if(Math.Abs(a2) <= 0.005)
84 {
85 if(Math.Abs(a1) <= 0.005)
86 {
87 // There are no roots found:
88 yield break;
89 } else {
90 // There is a single root, found from solving the linear equation with a1=0. We find a point close
91 // to the root using the linear approximation, after which we approach the true solution using the
92 // Newton-Raphson method:
93 QuadraticFunction f = new QuadraticFunction(a0, a1, a2);
94 yield return f.NewtonRaphson(-a0 / a1);
95
96 }
97 yield break;
98 }
99
100 double x1 = 0.5*(-a1 + Math.Sqrt(a1*a1 - 4.0*a0*a2))/a2;
101 double x2 = 0.5*(-a1 - Math.Sqrt(a1*a1 - 4.0*a0*a2))/a2;
102
103 if(Double.IsNaN(x1) == false)
104 {
105 yield return x1;
106 }
107
108 if(Double.IsNaN(x2) == false)
109 {
110 yield return x2;
