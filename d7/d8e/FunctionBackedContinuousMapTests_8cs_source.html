35 // Helper function for testing the implicit FunctionBackedContinuousMap constructor.
36 private static int TimesTwo(int x) => x * 2;
37
38 [Fact]
39 public void TestCasts()
40 {
41 // C# can't go straight from the function to a ContinuousMap, because technically it's already a cast to
42 // become a Func (or any delegate type), double implicit casts are not considered when looking for
43 // conversion operators, and there's no way to specify "method set" as the source for an implicit cast.
44 // Oh well. Still, if we explicitly cast to Func, our conversion works:
45 ContinuousMap<int, int> c = (Func<int, int>) TimesTwo;
46 Assert.NotNull(c);
47 Assert.Equal(10, c.GetValueAt(5));
48 Assert.IsAssignableFrom<FunctionBackedContinuousMap<int, int>>(c);
49 Assert.Equal(4, ((FunctionBackedContinuousMap<int, int>) c).F(2));
50
51 FunctionBackedContinuousMap<int, int> d = (Func<int, int>) TimesTwo;
52 Assert.NotNull(d);
53 Assert.Equal(10, d.GetValueAt(5));
54 Assert.Equal(4, d.F(2));
55 Assert.IsAssignableFrom<ContinuousMap<int, int>>(d);
56
57 var e = (ContinuousMap<int, int>) TimesTwo;
58 Assert.NotNull(e);
59 Assert.Equal(10, e.GetValueAt(5));
60 Assert.IsAssignableFrom<FunctionBackedContinuousMap<int, int>>(e);
61 Assert.Equal(4, ((FunctionBackedContinuousMap<int, int>) e).F(2));
62 }
63
64 [Fact]
65 public void TestThen()
66 {
67 ContinuousMap<int, int> c = ((ContinuousMap<int, int>)TimesTwo).Then(x => (x + 1));
68 Assert.NotNull(c);
69 Assert.Equal(expected: 15, actual: c.GetValueAt(7));
70 }
71
72 [Fact]
73 public void TestCompose()
74 {
75 ContinuousMap<int, int> c = ((ContinuousMap<int, int>)TimesTwo).Compose((int x) => x + 1);
76 Assert.NotNull(c);
77 Assert.Equal(16, c.GetValueAt(7));
78 }
