85 public override double GetNthDerivativeAt(double x, uint derivative)
86 {
87 // The input parameter must lie between the outer points, and must not be NaN:
88 if (!( x >= Points.Key[0] && x <= Points.Key[Points.Count - 1]))
89 {
90 throw new ArgumentOutOfRangeException(nameof(x), $"Cannot interpolate at {x}, which is outside the interval given by the spline points.");
91 }
92
93 // Find the index `i` of the closest point to the right of the input `x` parameter, which is the right point
94 // used to interpolate between. Therefore, `i-1` indicates the left point of the interval.
95 int i = Points.Key.BinarySearch(x);
96
97 // BinarySearch returns a bitwise complement of the index if the point is not exactly in the list, such as
98 // when interpolating. To turn it into a valid index, we take the bitwise complement again if it is negative:
99 if (i < 0)
100 {
101 i = ~i;
102 }
103
104 // If the index is zero, we are exactly on the first point in the list. We increment by one to get the value
105 // of the first spline segment, to avoid an IndexOutOfRangeException later on:
106 if (i == 0)
107 {
108 i++;
109 }
110
111 double x1 = Points.Key[i-1];
112 double x2 = Points.Key[i];
113 double y1 = Points.Value[i-1];
114 double y2 = Points.Value[i];
115
116 // Calculate and return the interpolated value:
117 double dx = x2 - x1;
118 double dy = y2 - y1;
119
120 double slope = dy / dx;
121
122 double localX = x - x1;
123
124 double localY = localX * slope;
125 double globalY = localY + y1;
126
127 // Return the result of evaluating a derivative function, depending on the derivative level:
128 switch (derivative)
129 {
130 case 0:
131 DebugUtil.AssertFinite(globalY, nameof(globalY));
132 return globalY;
133 case 1:
134 DebugUtil.AssertFinite(slope, nameof(slope));
135 return slope;
136 default: return 0.0;
137 }
138 }
