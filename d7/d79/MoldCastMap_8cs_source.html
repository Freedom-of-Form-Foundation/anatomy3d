139 public override double GetValueAt(dvec2 uv)
140 {
141 DebugUtil.AssertAllFinite(uv, nameof(uv));
142 double directionSign = (_direction == RayCastDirection.Outwards) ? 1.0 : -1.0;
143
144 Ray ray = new Ray(_raycastSurface.GetPositionAt(uv), directionSign*_raycastSurface.GetNormalAt(uv).Normalized);
145 double intersectionRadius = _moldSurface.RayIntersect(ray);
146
147 if (Math.Abs(intersectionRadius) <= _maxDistance)
148 {
149 return intersectionRadius;
150 } else {
151 return _defaultRadius.GetValueAt(uv);
152 }
153 }
