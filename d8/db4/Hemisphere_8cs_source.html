153 public override dvec3 GetNormalAt(dvec2 uv)
154 {
155 double u = uv.x;
156 double v = uv.y;
157
158 // Calculate the position of the rings of vertices:
159 double x = Math.Sin(v) * Math.Cos(u);
160 double y = Math.Sin(v) * Math.Sin(u);
161 double z = Math.Cos(v);
162
163 return x*Normal + y*Binormal + z*Direction;
164 }

167 public override dvec3 GetPositionAt(dvec2 uv)
168 {
169 dvec3 translation = Center;
170 double radius = this.Radius.GetValueAt(uv);
171
172 return translation + radius*GetNormalAt(uv);

176 public override List<Vertex> GenerateVertexList(int resolutionU, int resolutionV)
177 {
178 List<Vertex> output = new List<Vertex>(CalculateVertexCount(resolutionU, resolutionV));
179
180 // Load all required variables:
181 dvec3 pointTangent = Direction;
182 dvec3 pointNormal = Normal;
183 dvec3 pointBinormal = Binormal;
184
185 dvec3 translation = Center;
186
187 // Get the radius at the top of the hemisphere:
188 double topRadius = Radius.GetValueAt(new dvec2(0.0, 0.5 * Math.PI));
189
190 // Generate the first point at the pole of the hemisphere:
191 output.Add(new Vertex((vec3)(translation + topRadius * pointTangent), (vec3)pointTangent));
192
193 // Generate rings of the other points:
194 for (int j = 1; j < (resolutionV+1); j++)
195 {
196 for (int i = 0; i < resolutionU; i++)
197 {
198 // First find the normalized uv-coordinates, u = [0, 2pi], v = [0, 1/2pi]:
199 double u = i/(double)resolutionU * 2.0 * Math.PI;
200 double v = j/(double)resolutionV * 0.5 * Math.PI;
201
202 // Calculate the position of the rings of vertices:
203 double x = Math.Sin(v) * Math.Cos(u);
204 double y = Math.Sin(v) * Math.Sin(u);
205 double z = Math.Cos(v);
206
207 double radius = Radius.GetValueAt(new dvec2(u, v));
208
209 dvec3 surfaceNormal = x*pointNormal + y*pointBinormal + z*pointTangent;
210 dvec3 surfacePosition = translation + radius*surfaceNormal;
211
212 output.Add(new Vertex((vec3)surfacePosition, (vec3)surfaceNormal));
213 }
214 }
215
216 // Recalculate the surface normal after deformation:
217 for (int j = 1; j < resolutionV; j++)
218 {
219 for (int i = 0; i < (resolutionU - 1); i++)
220 {
221 dvec3 surfacePosition = output[(j-1)*resolutionU + i + 1].Position;
222 dvec3 du = surfacePosition - output[(j-1)*resolutionU + i + 1 + 1].Position;
223 dvec3 dv = surfacePosition - output[(j)*resolutionU + i + 1].Position;
224
225 // Calculate the position of the rings of vertices:
226 dvec3 surfaceNormal = dvec3.Cross(du.Normalized, dv.Normalized);
227
228 output[(j-1)*resolutionU + i + 1] = new Vertex((vec3)surfacePosition, (vec3)surfaceNormal);
229 }
230
231 // Stitch the end of the triangles:
232 dvec3 surfacePosition2 = output[(j-1)*resolutionU + resolutionU].Position;
233 dvec3 du2 = surfacePosition2 - output[(j-1)*resolutionU + 1].Position;
234 dvec3 dv2 = surfacePosition2 - output[(j)*resolutionU + resolutionU].Position;
235
236 // Calculate the position of the rings of vertices:
237 dvec3 surfaceNormal2 = dvec3.Cross(du2.Normalized, dv2.Normalized);
238
239 output[(j-1)*resolutionU + resolutionU] = new Vertex((vec3)surfacePosition2, (vec3)surfaceNormal2);
240 }
241
242 return output;
243 }

246 public override List<int> GenerateIndexList(int resolutionU, int resolutionV, int indexOffset = 0)
247 {
248 List<int> output = new List<int>(CalculateIndexCount(resolutionU, resolutionV));
249
250 // Add a triangle fan on the pole of the hemisphere:
251 for (int i = 0; i < resolutionU-1; i++)
252 {
253 output.Add(indexOffset + 1 + i);
254 output.Add(indexOffset + 2 + i);
255 output.Add(indexOffset + 0);
256 }
257
258 // Stitch final triangle on the pole of the hemisphere:
259 output.Add(indexOffset + resolutionU);
260 output.Add(indexOffset + 1);
261 output.Add(indexOffset + 0);
262
263 // Add the remaining rings:
264 for (int j = 0; j < resolutionV - 1; j++)
265 {
266 // Add a ring of triangles:
267 for (int i = 0; i < resolutionU-1; i++)
268 {
269 output.Add(indexOffset + 1 + i + resolutionU*(j+1));
270 output.Add(indexOffset + 1 + (i+1) + resolutionU*j);
271 output.Add(indexOffset + 1 + i + resolutionU*j);
272
273 output.Add(indexOffset + 1 + (i+1) + resolutionU*(j+1));
274 output.Add(indexOffset + 1 + (i+1) + resolutionU*j);
275 output.Add(indexOffset + 1 + i + resolutionU*(j+1));
276 }
277
278 // Stitch the end of the ring of triangles:
279 output.Add(indexOffset + 1 + resolutionU-1 + resolutionU*(j+1));
280 output.Add(indexOffset + 1 + resolutionU*j);
281 output.Add(indexOffset + 1 + resolutionU-1 + resolutionU*j);
282
283 output.Add(indexOffset + 1 + resolutionU*(j+1));
284 output.Add(indexOffset + 1 + resolutionU*j);
285 output.Add(indexOffset + 1 + resolutionU-1 + resolutionU*(j+1));
286 }
287
288 return output;
289 }
