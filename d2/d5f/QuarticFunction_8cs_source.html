50 public override double GetNthDerivativeAt(double x, uint derivative)
51 {
52 DebugUtil.AssertFinite(x, nameof(x));
53 // Return a different function depending on the derivative level:
54 switch (derivative)
55 {
56 case 0: return A0 + A1*x + A2*x*x + A3*x*x*x + A4*x*x*x*x;
57 case 1: return A1 + 2.0*A2*x + 3.0*A3*x*x + 4.0*A4*x*x*x;
58 case 2: return 2.0*A2 + 6.0*A3*x + 12.0*A4*x*x;
59 case 3: return 6.0*A3 + 24.0*A4*x;
60 case 4: return 24.0*A4;
61 default: return 0.0;
62 }
63 }

81 public static IEnumerable<double> Solve(double a0, double a1, double a2, double a3, double a4)
82 {
83 DebugUtil.AssertAllFinite(new double[]{a0, a1, a2, a3, a4}, "a");
84 if(Math.Abs(a4) <= 0.005)
85 {
86 foreach (double v in CubicFunction.Solve(a0, a1, a2, a3))
87 {
88 QuarticFunction f = new QuarticFunction(a0, a1, a2, a3, a4);
89 yield return f.NewtonRaphson(v);
90 }
91 yield break;
92 }
93
94 double ba = a3/a4;
95 double ca = a2/a4;
96 double da = a1/a4;
97 // double ea = a0/a4;
98
99 double p1 = a2*a2*a2 - 4.5*a3*a2*a1 + 13.5*a4*a1*a1 + 13.5*a3*a3*a0 - 36.0*a4*a2*a0;
100
101 double q = a2*a2 - 3.0*a3*a1 + 12.0*a4*a0;
102
103 Complex p2 = p1 + Complex.Sqrt(-q*q*q + p1*p1);
104
105 Complex pow = Complex.Pow(p2, (1.0/3.0));
106
107 Complex p3 = q/(3.0*a4*pow) + pow/(3.0*a4);
108
109 Complex p4 = Complex.Sqrt(ba*ba/4.0 - 2.0*ca/(3.0) + p3);
110
111 Complex p5 = a3*a3/(2.0*a4*a4) - 4.0*a2/(3.0*a4) - p3;
112
113 Complex p6 = (-ba*ba*ba + 4.0*ba*ca - 8.0*da)/(4.0*p4);
114
115 List<Complex> roots = new List<Complex>
116 {
117 -ba/(4.0) - p4/2.0 - 0.5*Complex.Sqrt(p5 - p6),
118 -ba/(4.0) - p4/2.0 + 0.5*Complex.Sqrt(p5 - p6),
119 -ba/(4.0) + p4/2.0 - 0.5*Complex.Sqrt(p5 + p6),
120 -ba/(4.0) + p4/2.0 + 0.5*Complex.Sqrt(p5 + p6),
121 };
122
123 foreach (Complex root in roots)
124 {
125 if(Math.Abs(root.Imaginary) <= 0.005)
126 {
127 DebugUtil.AssertFinite(root.Real, nameof(root.Real));
128 yield return root.Real;
129 }
130 }
131 }
