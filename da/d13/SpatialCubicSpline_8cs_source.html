67 public SpatialCubicSpline(SortedList<double, dvec3> points)
68 {
69 // Note: slow implementation, can be optimized in C++.
70 SortedList<double, double> x = new SortedList<double, double>();
71 SortedList<double, double> y = new SortedList<double, double>();
72 SortedList<double, double> z = new SortedList<double, double>();
73
74 foreach (var item in points)
75 {
76 x.Add(item.Key, item.Value.x);
77 y.Add(item.Key, item.Value.y);
78 z.Add(item.Key, item.Value.z);
79 }
80
81 X = new CubicSpline1D(x);
82 Y = new CubicSpline1D(y);
83 Z = new CubicSpline1D(z);
84
85 GenerateCurveNormal();
86 }

105 public override dvec3 GetNormalAt(double t)

126#region Private Methods
133 private void GenerateCurveNormal()
134 {
135 // Generate a tangent at key points:
136 // TODO: make the precision adapt to the precision of centerCurve!
137
138 SortedList<double, double> normalX = new SortedList<double, double>();
139 SortedList<double, double> normalY = new SortedList<double, double>();
140 SortedList<double, double> normalZ = new SortedList<double, double>();
141
142 // No normal supplied, pick arbitrary normal vector and calculate a binormal:
143 dvec3 binormal = dvec3.Cross(VectorUtilities.InventNormal(GetTangentAt(0.0)), GetTangentAt(0.0));
144
145 // Traverse through all the points in the spline, and calculate the next normal, maintaining smoothness by
146 // memorizing the previous normal:
147
148 double previousValue = X.Points.Key[0]; // Location.Points.Key is guaranteed to have at least 2 entries, this is safe.
149 for (int i = 1; i < X.Points.Count; i++)
150 {
151 double currentValue = X.Points.Key[i];
152 double segmentSize = currentValue - previousValue;
153
154 // Interpolate between each two points:
155 for (int j = 0; j < kInterpolationStepCount; j++)
156 {
157 double t = previousValue + segmentSize*j/kInterpolationStepCount;
158
159 dvec3 direction = GetTangentAt(t).Normalized;
160 dvec3 normal = dvec3.Cross(direction, binormal).Normalized;
161 binormal = dvec3.Cross(normal, direction);
162
163 normalX.Add(t, normal.x);
164 normalY.Add(t, normal.y);
165 normalZ.Add(t, normal.z);
166 }
167
168 previousValue = currentValue;
169 }
170
171 // Add the final point as well:
172 {
173 double t = X.Points.Key[X.Points.Count - 1];
174
175 dvec3 direction = GetTangentAt(t).Normalized;
176 dvec3 normal = dvec3.Cross(direction, binormal).Normalized;
177 binormal = dvec3.Cross(normal, direction);
178
179 normalX.Add(t, normal.x);
180 normalY.Add(t, normal.y);
181 normalZ.Add(t, normal.z);
182 }
183
184 NormalX = new CubicSpline1D(normalX);
185 NormalY = new CubicSpline1D(normalY);
186 NormalZ = new CubicSpline1D(normalZ);
187 }
188#endregion Private Methods
189 }
