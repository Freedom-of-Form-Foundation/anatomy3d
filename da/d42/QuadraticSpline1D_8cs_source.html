53 public QuadraticSpline1D(SortedList<double, double> points)
54 {
55 if (points.Count < 2)
56 {
57 if (points.Count == 1)
58 {
59 throw new ArgumentException("List contains only a single point. A spline must have at least two points.", nameof(points));
60 }
61 else
62 {
63 throw new ArgumentException("List is empty. A spline must have at least two points.", nameof(points));
64 }
65 }
66
67 Points = new SortedPointsList<double>(points);
68
69 // Calculate the coefficients of the spline:
70 _parameters = new double[points.Count];
71
72 // Find the first segment parameter, which will be a straight line:
73 {
74 double dx = Points.Key[1] - Points.Key[0];
75 double dy = Points.Value[1] - Points.Value[0];
76
77 _parameters[0] = dy/dx;
78 }
79
80 // Recursively find the other _parameters:
81 for (int i = 1; i < points.Count; i++)
82 {
83 double dx = Points.Key[i] - Points.Key[i-1];
84 double dy = Points.Value[i] - Points.Value[i-1];
85
86 _parameters[i] = -_parameters[i-1] + 2.0f*dy/dx;
87 }
88
89 }

107 public override double GetNthDerivativeAt(double x, uint derivative)
108 {
109 // The input parameter must lie between the outer points, and must not be NaN:
110 if (!( x >= Points.Key[0] && x <= Points.Key[Points.Count - 1]))
111 {
112 throw new ArgumentOutOfRangeException(nameof(x), "Cannot interpolate outside the interval given by the spline points.");
113 }
114
115 // Find the index `i` of the closest point to the right of the input `x` parameter, which is the right point
116 // used to interpolate between. Therefore, `i-1` indicates the left point of the interval.
117 int i = Points.Key.BinarySearch(x);
118
119 // BinarySearch returns a bitwise complement of the index if the point is not exactly in the list, such as
120 // when interpolating. To turn it into a valid index, we take the bitwise complement again if it is negative:
121 if (i < 0)
122 {
123 i = ~i;
124 }
125
126 // If the index is zero, we are exactly on the first point in the list. We increment by one to get the value
127 // of the first spline segment, to avoid an IndexOutOfRangeException later on:
128 if (i == 0)
129 {
130 i++;
131 }
132
133 double x1 = Points.Key[i-1];
134 double x2 = Points.Key[i];
135 double y1 = Points.Value[i-1];
136 double y2 = Points.Value[i];
137
138 // Calculate and return the interpolated value:
139 double dx = x2 - x1;
140 double dy = y2 - y1;
141
142 double a = -_parameters[i]*dx + dy;
143 double t = (x-x1)/dx;
144
145 // Return a different function depending on the derivative level:
146 switch (derivative)
147 {
148 case 0: return (1.0 - t)*y1 + t*y2 + t*(1.0-t)*a;
149 case 1: return (dy + a * (1.0 - 2.0 * t))/dx;
150 case 2: return (2.0 * a * t)/(dx*dx);
151 default: return 0.0;
152 }
153 }

162 public override double GetValueAt(double x)
163 {
164 return GetNthDerivativeAt(x, 0);
165 }

172 public override IEnumerable<double> SolveRaytrace(QuarticFunction surfaceFunction, double z0 = 0.0, double c = 1.0)
173 {
174 // Solve the polynomial equation for each segment:
175 for (int i = 1; i < Points.Count; i++)
176 {
177 double x1 = Points.Key[i-1];
178 double x2 = Points.Key[i];
179 double y1 = Points.Value[i-1];
180 double y2 = Points.Value[i];
181
182 // Calculate and return the interpolated value:
183 double dx = x2 - x1;
184 double div = 1.0/dx;
185 double dy = y2 - y1;
186
187 double a = -_parameters[i]*dx + dy;
188
189 // Write in the form of a0 + a1 z + a2 z^2:
190 double a0 = -a*x1*x1*div*div - (a + dy)*x1*div + y1;
191 double a1 = 2.0*a*x1*div*div + (a + dy)*div;
192 double a2 = -a*div*div;
193
194 // Substitute z = z0 + c t:
195 double q0 = a0 + a1*z0 + a2*z0*z0;
196 double q1 = (a1 + 2.0*a2*z0)*c;
197 double q2 = a2*c*c;
198
199 // Find the quartic polynomial to solve:
200 double p0 = surfaceFunction.A0 - q0*q0;
201 double p1 = surfaceFunction.A1 - 2.0*q0*q1;
202 double p2 = surfaceFunction.A2 - (2.0*q0*q2 + q1*q1);
203 double p3 = surfaceFunction.A3 - 2.0*q1*q2;
204 double p4 = surfaceFunction.A4 - q2*q2;
205
206 // Solve the quartic polynomial:
207 IEnumerable<double> intersections = QuarticFunction.Solve(p0, p1, p2, p3, p4);
208
209 // Only return the value if it is sampled within the segment that we are currently considering,
210 // otherwise the value we got is invalid:
211 foreach (var j in intersections)
212 {
213 if (((z0 + c*j) > x1) && ((z0 + c*j) <= x2))
214 {
215 yield return j;
216 }
217 }
218 }
219 }
