68 [Fact]
69 public void TestVaryingSort()
70 {
71 int countA = 0;
72 MovingPoint2D pA = new MovingPoint2D(() => { countA++; }, 0, 0);
73
74 int countB = 0;
75 MovingPoint2D pB = new MovingPoint2D(() => { countB++; }, 1, 1);
76
77 int countC = 0;
78 MovingPoint2D pC = new MovingPoint2D(() => { countC++; }, 2, 2);
79
80 List<MovingPoint2D> list = new List<MovingPoint2D> {pC, pB, pA};
81 Assert.Equal(new List<MovingPoint2D>{pC, pB, pA}, list);
82 list.Sort();
83 Assert.Equal(new List<MovingPoint2D>{pA, pB, pC}, list);
84 Assert.Equal(0, countA);
85 Assert.Equal(0, countB);
86 Assert.Equal(0, countC);
87
88 pA.X = 3;
89 list.Sort();

92 pB.X = 3;

95
96 pA.Y = 5;
